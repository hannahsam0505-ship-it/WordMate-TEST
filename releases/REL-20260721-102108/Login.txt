      const loginApiElapsedMs = Math.round(performance.now() - loginStartedAt);

      if (data && data.requireConfirm === true) {
        wmPendingLogin = { studentId, password };
        wmShowForceLoginModal(data.message || "기존 접속은 종료됩니다.");

      wmPendingLogin = null;

      msg.style.color = "green";

      const currentSet = s.현재세트 || wmBuildInitialSetId(s.학습배정);

          현재세션: sessionToken || "",
          sessionToken: sessionToken || "",
          wmSessionToken: sessionToken || ""

  function wmBuildInitialSetId(learningAssign) {
    const text = String(learningAssign || "");
    const match = text.match(/(\d+)/);
    if (!match) return "5-1-1";
    return Number(match[1]) + "-1-1";
  }

  function wmNormalizeSetIdForStudy(setId) {
    const text = String(setId || "").trim().toUpperCase();

    if (/^WM\d+-\d+-\d+$/.test(text)) {
      return text;
    }

    if (/^\d+-\d+-\d+$/.test(text)) {
      return "WM" + text;
    }

    return "WM4-1-1";
  }
